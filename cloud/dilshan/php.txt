Install PHP 5.6.x
================================
add-apt-repository ppa:ondrej/php

---remove ppa -----
sudo add-apt-repository --remove ppa:whatever/ppa
-------------------

apt-get update
apt-get install php5.6

apt-get install php5.6-mysql php5.6-curl php5.6-gd php5.6-intl php-pear php5.6-imagick php5.6-imap php5.6-mcrypt php5.6-memcache

sudo apt-get install php5.6 php5.6-mbstring php5.6-mcrypt php5.6-mysql php5.6-xml
sudo a2dismod php5 php-mbstring php-gettext
sudo a2enmod php5.6
